  290.     /**

  292.      * @param string $sourceVer The source version number.
  293.      * @param string $compareVer The version number to compare with the source version number.

  298.     {
  299.         $sourceVer = explode('.', $sourceVer);
  300.         foreach ($sourceVer as $k => $v) {
  301.             $sourceVer[$k] = $this->parseInt($v);
  302.         }
  303.  
  304.         $compareVer = explode('.', $compareVer);
  305.         foreach ($compareVer as $k => $v) {
  306.             $compareVer[$k] = $this->parseInt($v);
  307.         }
  308.  
  309.         if (count($sourceVer) != count($compareVer)) {
  310.             if (count($sourceVer) > count($compareVer)) {
  311.                 for ($i = count($compareVer); $i < count($sourceVer); $i++) {
  312.                     $compareVer[$i] = 0;
  313.                 }
  314.             } else {
  315.                 for ($i = count($sourceVer); $i < count($compareVer); $i++) {
  316.                     $sourceVer[$i] = 0;
  317.                 }
  318.             }
  319.         }
  320.  
  321.         foreach ($sourceVer as $i => $srcVerPart) {
  322.             if ($srcVerPart > $compareVer[$i]) {
  323.                 return 1;
  324.             } else {
  325.                 if ($srcVerPart < $compareVer[$i]) {
  326.                     return -1;
  327.                 }
  328.             }
  329.         }
  330.  
  331.         return 0;
  332.     }

  388.      * systems is supported.
  389.      * @param boolean $returnVersionNumbers Determines if the return value must be versions numbers as a string (true)
  390.      *  or the version name (false).
  391.      * @param boolean $returnServerFlavor Since some Windows NT versions have the same values, this flag determines if
  392.      *  the Server flavor is returned or not. For instance Windows 8.1 and Windows Server 2012 R2 both use version 6.3.
  393.      * @return string Returns the version name/version numbers of the platform or the constant PLATFORM_VERSION_UNKNOWN
  394.      *  if unknown.
  395.      */
  396.     public function getPlatformVersion($returnVersionNumbers = false, $returnServerFlavor = false)
  397.     {
  398.         if ($this->_platformVersion == self::PLATFORM_VERSION_UNKNOWN || $this->_platformVersion == '') {
  399.             return self::PLATFORM_VERSION_UNKNOWN;
  400.         }
  401.  
  402.         if ($returnVersionNumbers) {
  403.             return $this->_platformVersion;
  404.         } else {
  405.             switch ($this->getPlatform()) {
  406.                 case self::PLATFORM_WINDOWS:
  407.                     if (substr($this->_platformVersion, 0, 3) == 'NT ') {
  408.                         return $this->windowsNTVerToStr(substr($this->_platformVersion, 3), $returnServerFlavor);
  409.                     } else {
  410.                         return $this->windowsVerToStr($this->_platformVersion);
  411.                     }
  412.                 break;
  413.  
  414.                 default: return self::PLATFORM_VERSION_UNKNOWN;
  415.             }
  416.         }
  417.     }

  710.      * @link http://en.wikipedia.org/wiki/Internet_Explorer_Mobile
  711.      * @return boolean Returns true if the browser is Internet Explorer, false otherwise.
  712.      */
  713.     protected function checkBrowserInternetExplorer()
  714.     {
  715.         //Test for Internet Explorer Mobile (formerly Pocket Internet Explorer)
  716.         if ($this->checkSimpleBrowserUA(array('IEMobile', 'MSPIE'), $this->_agent, self::BROWSER_IE_MOBILE, true)) {
  717.             return true;
  718.         }
  719.  
  720.         //Several browsers uses IE compatibility UAs filter these browsers out (but after testing for IE Mobile)
  721.         if (stripos($this->_agent, 'Opera') !== false ||
  722.                 stripos($this->_agent, 'BlackBerry') !== false ||
  723.                 stripos($this->_agent, 'Nokia') !== false) {
  724.             return false;
  725.         }
  726.  
  727.         //Test for Internet Explorer 1
  728.         if ($this->checkSimpleBrowserUA('Microsoft Internet Explorer', $this->_agent, self::BROWSER_IE)) {
  729.             if ($this->getVersion() == self::VERSION_UNKNOWN) {
  730.                 if (preg_match('/308|425|426|474|0b1/i', $this->_agent)) {
  731.                     $this->setVersion('1.5');
  732.                 } else {
  733.                     $this->setVersion('1.0');
  734.                 }
  735.             }
  736.             return true;
  737.         }
  738.  
  739.         //Test for Internet Explorer 2+
  740.         if (stripos($this->_agent, 'MSIE') !== false || stripos($this->_agent, 'Trident') !== false) {
  741.             $version = '';
  742.  
  743.             if (stripos($this->_agent, 'Trident') !== false) {
  744.                 //Test for Internet Explorer 11+ (check the rv: string)
  745.                 if (stripos($this->_agent, 'rv:') !== false) {
  746.                     if ($this->checkSimpleBrowserUA('Trident', $this->_agent, self::BROWSER_IE, false, false, 'rv:')) {
  747.                         return true;
  748.                     }
  749.                 } else {
  750.                     //Test for Internet Explorer 8, 9 & 10 (check the Trident string)
  751.                     if (preg_match('/Trident\/([\d]+)/i', $this->_agent, $foundVersion)) {
  752.                         //Trident started with version 4.0 on IE 8
  753.                         $verFromTrident = $this->parseInt($foundVersion[1]) + 4;
  754.                         if ($verFromTrident >= 8) {
  755.                             $version = $verFromTrident . '.0';
  756.                         }
  757.                     }
  758.                 }
  759.  
  760.                 //If we have the IE version from Trident, we can check for the compatibility view mode
  761.                 if ($version != '') {
  762.                     $emulatedVer = '';
  763.                     preg_match_all('/MSIE\s*([^\s;$]+)/i', $this->_agent, $foundVersions);
  764.                     foreach ($foundVersions[1] as $currVer) {
  765.                         //Keep the lowest MSIE version for the emulated version (in compatibility view mode)
  766.                         if ($emulatedVer == '' || $this->compareVersions($emulatedVer, $currVer) == 1) {
  767.                             $emulatedVer = $currVer;
  768.                         }
  769.                     }
  770.                     //Set the compatibility view mode if $version != $emulatedVer
  771.                     if ($this->compareVersions($version, $emulatedVer) != 0) {
  772.                         $this->_compatibilityViewName = self::BROWSER_IE;
  773.                         $this->_compatibilityViewVer = $this->cleanVersion($emulatedVer);
  774.                     }
  775.                 }
  776.             }
  777.  
  778.             //Test for Internet Explorer 2-7 versions if needed
  779.             if ($version == '') {
  780.                 preg_match_all('/MSIE\s+([^\s;$]+)/i', $this->_agent, $foundVersions);
  781.                 foreach ($foundVersions[1] as $currVer) {
  782.                     //Keep the highest MSIE version
